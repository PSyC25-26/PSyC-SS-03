204 private String extraerValor(String texto, String clave) {
205 String[] pares = texto.split(",");
206
207 for (String par : pares) {
208 String[] partes = par.split(":", 2);
209 if (partes.length == 2) {
210 String key = partes[0].replace("\"", "").trim();
211 String value = partes[1].replace("\"", "").trim();
212
213 if (key.equals(clave)) {
214 return value;
215 }
216 }
217 }
218 return "";
