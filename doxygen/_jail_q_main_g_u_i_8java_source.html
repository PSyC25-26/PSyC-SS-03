132 private JLabel cargarLogo(int ancho, int alto) {
133 try {
134 var stream = getClass().getResourceAsStream("/assets/jailqualitylogobase.png");
135 if (stream != null) {
136 BufferedImage original = ImageIO.read(stream);
137 BufferedImage scaled = new BufferedImage(ancho, alto, BufferedImage.TYPE_INT_ARGB);
138 Graphics2D g2 = scaled.createGraphics();
139 g2.setRenderingHint(RenderingHints.KEY_INTERPOLATION, RenderingHints.VALUE_INTERPOLATION_BICUBIC);
140 g2.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
141 g2.drawImage(original, 0, 0, ancho, alto, null);
142 g2.dispose();
143 return new JLabel(new ImageIcon(scaled));
144 }
145 } catch (Exception ignored) { }
146 JLabel sustituto = new JLabel("JQ");
147 sustituto.setFont(new Font("Segoe UI", Font.BOLD, 36));
148 return sustituto;
149 }

449 private void comprobarConexionServidor() {
450 lblEstado.setText("Comprobando...");
451 lblEstado.setForeground(Color.DARK_GRAY);
452
453 new Thread(() -> {
454 boolean ok = false;
455 try {
456 HttpRequest req = HttpRequest.newBuilder()
457 .uri(URI.create("http://localhost:8080/carcel"))
458 .GET().build();
459 int codigo = HttpClientSingleton.getInstance().getClient()
460 .send(req, HttpResponse.BodyHandlers.discarding()).statusCode();
461 ok = codigo >= 200 && codigo < 300;
462 } catch (Exception ignored) { }
463
464 boolean conectado = ok;
465 SwingUtilities.invokeLater(() -> {
466 lblEstado.setFont(new Font("Dialog", Font.PLAIN, 14));
467 lblEstado.setText(conectado ? "✓ Conectado" : "✗ Sin conexión");
468 lblEstado.setForeground(conectado ? new Color(0, 128, 0) : new Color(180, 30, 30));
469 });
470 }).start();
471 }
