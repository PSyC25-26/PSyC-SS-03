86 public Preso anadirPreso(Preso nuevoPreso) {

88 if (nuevoPreso == null) {

92 LocalDate fechaActual = LocalDate.now();

94 // Validación de nombre
95 if (nuevoPreso.getNombre() == null || nuevoPreso.getNombre().trim().isEmpty()) {
96 throw new IllegalArgumentException("El nombre es obligatorio.");
97 }
98
99 // Validación de apellidos
100 if (nuevoPreso.getApellidos() == null || nuevoPreso.getApellidos().trim().isEmpty()) {
101 throw new IllegalArgumentException("Los apellidos son obligatorios.");
102 }
103
104 // Validación de fecha de nacimiento
105 if (nuevoPreso.getFechaNacimiento() == null) {
106 throw new IllegalArgumentException("La fecha de nacimiento es obligatoria.");
107 }

109 if (nuevoPreso.getFechaNacimiento().plusYears(18).isAfter(fechaActual)) {

112

130

136 // 1. Buscamos la cárcel completa en la base de datos usando el ID

139
140 // 2. Asignamos la cárcel completa (con nombre, localidad, etc.) al preso

145 // 3. Guardamos y devolvemos el objeto (ahora con todos los datos)
